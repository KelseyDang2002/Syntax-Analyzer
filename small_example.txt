[* Function to calculate the factorial of an integer using recursion *]
function factorial(n integer) {
    if (n <= 1) {
        ret 1;  [* Base case: factorial of 0 or 1 is 1 *]
    } else {
        result = n * factorial;  [* Recursive case: multiply by (n-1)! *]
        ret result;
    } endif
}
#
    [* Main program *]
    factorial(5);
#
